cmake_minimum_required(VERSION 3.16)

project(Agnes VERSION 0.1.0 LANGUAGES CXX)

# ---- Project options ---------------------------------------------------------
option(AGNES_ENABLE_WARNINGS "Enable stricter compiler warnings" ON)
option(AGNES_BUILD_TESTS "Enable test scaffolding" OFF)

# ---- Global language settings ------------------------------------------------
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# ---- Target layout -----------------------------------------------------------
set(AGNES_TARGET Agnes)
set(AGNES_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR}/src)

set(AGNES_SOURCES
	${AGNES_SRC_DIR}/main.cpp
	${AGNES_SRC_DIR}/ppm.cpp
	${AGNES_SRC_DIR}/cli.cpp
)

set(AGNES_HEADERS
	${AGNES_SRC_DIR}/ppm.h
	${AGNES_SRC_DIR}/filepath.h
	${AGNES_SRC_DIR}/color.h
	${AGNES_SRC_DIR}/vec.h
	${AGNES_SRC_DIR}/agnes_math.h
	${AGNES_SRC_DIR}/cli.h
)

add_executable(${AGNES_TARGET})

target_sources(${AGNES_TARGET}
	PRIVATE
		${AGNES_SOURCES}
		${AGNES_HEADERS}
)

target_include_directories(${AGNES_TARGET}
	PRIVATE
		${AGNES_SRC_DIR}
)

target_compile_definitions(${AGNES_TARGET}
	PRIVATE
		AGNES_PROJECT_SOURCE_DIR="${CMAKE_CURRENT_SOURCE_DIR}"
)

# Keep source files grouped in IDEs (Visual Studio/Xcode).
source_group(TREE ${AGNES_SRC_DIR} FILES ${AGNES_SOURCES} ${AGNES_HEADERS})

if(AGNES_ENABLE_WARNINGS)
	if(MSVC)
		target_compile_options(${AGNES_TARGET} PRIVATE /W4 /permissive-)
	else()
		target_compile_options(${AGNES_TARGET} PRIVATE -Wall -Wextra -Wpedantic)
	endif()
endif()

set_target_properties(${AGNES_TARGET}
	PROPERTIES
		RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)

# Placeholder to keep test integration straightforward when tests are added.
if(AGNES_BUILD_TESTS)
	include(CTest)
	if(BUILD_TESTING)
		message(STATUS "Tests are enabled, but no test targets are defined yet.")
	endif()
endif()
